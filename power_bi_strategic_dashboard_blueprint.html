Core DAX measures — production-ready
Revenue & Profit
// ── Total Revenue ────────────────────────────────────── Total Revenue = CALCULATE( SUM( FactSales[SaleAmount] ), ALL( DimDate[IsHoliday] ) ) // ── Total Profit ─────────────────────────────────────── Total Profit = [Total Revenue] - SUM( FactSales[COGS] ) - CALCULATE( SUM( FactFinance[ActualAmount] ), FactFinance[GLCategory] = "OpEx" ) // ── Profit Margin % ──────────────────────────────────── Profit Margin % = DIVIDE( [Total Profit], [Total Revenue], 0 ) // ── Revenue Growth % ─────────────────────────────────── Revenue Growth % = VAR CY = [Total Revenue] VAR PY = CALCULATE( [Total Revenue], SAMEPERIODLASTYEAR( DimDate[FullDate] ) ) RETURN DIVIDE( CY - PY, PY, 0 ) // ── CAGR (3-Year) ────────────────────────────────────── CAGR 3Y = VAR EndVal = [Total Revenue] VAR BegVal = CALCULATE( [Total Revenue], DATESINPERIOD( DimDate[FullDate], MIN(DimDate[FullDate]), -3, YEAR ) ) RETURN POWER( DIVIDE( EndVal, BegVal, 1 ), 1/3 ) - 1 // ── EBITDA ───────────────────────────────────────────── EBITDA = [Total Profit] + CALCULATE( SUM( FactFinance[ActualAmount] ), FactFinance[GLCategory] IN { "Depreciation", "Amortization", "Interest", "Tax" } )
Budget, Forecast & ROI
// ── Budget Variance ──────────────────────────────────── Budget Variance = [Total Revenue] - SUM( FactTargets[TargetValue] ) Budget Variance % = DIVIDE( [Budget Variance], SUM( FactTargets[TargetValue] ), 0 ) // ── Forecast Accuracy ────────────────────────────────── Forecast Accuracy % = 1 - ABS( DIVIDE( [Total Revenue] - SUM( FactFinance[ForecastAmount] ), SUM( FactFinance[ForecastAmount] ), 0 ) ) // ── ROI ──────────────────────────────────────────────── ROI % = VAR TotalInvest = CALCULATE( SUM( FactFinance[ActualAmount] ), FactFinance[GLCategory] = "Investment" ) RETURN DIVIDE( [Total Profit] - TotalInvest, TotalInvest, 0 )
Customer Metrics
// ── Customer Retention % ─────────────────────────────── Customer Retention % = VAR StartCust = CALCULATE( DISTINCTCOUNT( FactCustomer[CustomerKey] ), DATESINPERIOD( DimDate[FullDate], FIRSTDATE( DimDate[FullDate] ), -1, YEAR ) ) VAR ChurnCust = CALCULATE( COUNTROWS( FactCustomer ), FactCustomer[ChurnFlag] = 1 ) RETURN DIVIDE( StartCust - ChurnCust, StartCust, 0 ) // ── Churn % ──────────────────────────────────────────── Churn % = 1 - [Customer Retention %] // ── Customer LTV ─────────────────────────────────────── Customer LTV = VAR AvgRevPerCust = DIVIDE( [Total Revenue], DISTINCTCOUNT( FactSales[CustomerKey] ), 0 ) VAR AvgLifespan = DIVIDE( 1, [Churn %], 0 ) VAR GrossMargin = [Profit Margin %] RETURN AvgRevPerCust * AvgLifespan * GrossMargin // ── Customer Acquisition Cost ────────────────────────── Customer Acquisition Cost = VAR MktgSales = CALCULATE( SUM( FactFinance[ActualAmount] ), FactFinance[GLCategory] IN { "Marketing", "Sales" } ) VAR NewCusts = CALCULATE( COUNTROWS( FactCustomer ), FactCustomer[EventType] = "New" ) RETURN DIVIDE( MktgSales, NewCusts, 0 ) // ── NPS Score ────────────────────────────────────────── NPS Score = VAR Promoters = CALCULATE( COUNTROWS( FactCustomer ), FactCustomer[NPS_Score] >= 9 ) VAR Detractors = CALCULATE( COUNTROWS( FactCustomer ), FactCustomer[NPS_Score] <= 6 ) VAR Total = COUNTROWS( FactCustomer ) RETURN DIVIDE( Promoters - Detractors, Total, 0 ) * 100
Strategic Indices
// ── Market Share % ───────────────────────────────────── Market Share % = DIVIDE( [Total Revenue], CALCULATE( [Total Revenue], ALL( DimCustomer ) ) * 4.2, 0 ) // Note: 4.2 = estimated total addressable market multiplier // ── Strategic Performance Index ──────────────────────── Strategic Performance Index = VAR w_Rev = [Revenue Growth %] * 25 VAR w_Marg = [Profit Margin %] * 20 VAR w_Ret = [Customer Retention %] * 20 VAR w_NPS = ( [NPS Score] + 100 ) / 200 * 15 VAR w_ROI = MIN( [ROI %] / 0.4, 1 ) * 10 VAR w_Ops = AVERAGE( FactOperations[EfficiencyScore] ) / 100 * 10 RETURN w_Rev + w_Marg + w_Ret + w_NPS + w_ROI + w_Ops // ── Business Health Score ────────────────────────────── Business Health Score = VAR FinHealth = MIN( [Profit Margin %] / 0.25, 1 ) * 30 VAR CustHealth = [Customer Retention %] * 25 VAR OpsHealth = AVERAGE( FactOperations[EfficiencyScore] ) / 100 * 25 VAR GrowHealth = MIN( [Revenue Growth %] / 0.20, 1 ) * 20 RETURN ( FinHealth + CustHealth + OpsHealth + GrowHealth ) * 100 // ── Productivity Score ───────────────────────────────── Productivity Score = DIVIDE( [Total Revenue], CALCULATE( SUM( DimDepartment[HeadCount] ), ALL( DimDepartment ) ), 0 )

Date Table Generation
// DimDate — fully computed calendar table let StartDate = #date(2020, 1, 1), EndDate = Date.From(DateTime.LocalNow()), DateRange = {Number.From(StartDate)..Number.From(EndDate)}, DateList = List.Transform(DateRange, Number.ToDate), ToTable = Table.FromList(DateList, Splitter.SplitByNothing(), {"Date"}, null, ExtraValues.Error), AddYear = Table.AddColumn(ToTable, "Year", each Date.Year([Date]), Int64.Type), AddQtr = Table.AddColumn(AddYear, "Quarter", each "Q" & Text.From(Date.QuarterOfYear([Date])), type text), AddMonth = Table.AddColumn(AddQtr, "MonthNum", each Date.Month([Date]), Int64.Type), AddMonthNm = Table.AddColumn(AddMonth, "MonthName", each Date.MonthName([Date]), type text), AddWeek = Table.AddColumn(AddMonthNm, "WeekNum", each Date.WeekOfYear([Date]), Int64.Type), AddDoW = Table.AddColumn(AddWeek, "DayOfWeek", each Date.DayOfWeekName([Date]), type text), AddIsWknd = Table.AddColumn(AddDoW, "IsWeekend", each Date.DayOfWeek([Date]) >= 5, type logical), AddFiscalYr = Table.AddColumn(AddIsWknd, "FiscalYear", each if Date.Month([Date]) >= 7 then Date.Year([Date]) + 1 else Date.Year([Date]), Int64.Type), AddDateKey = Table.AddColumn(AddFiscalYr, "DateKey", each Int32.From(Date.ToText([Date], "yyyyMMdd")), Int64.Type), SetTypes = Table.TransformColumnTypes(AddDateKey, {{"Date", type date}}) in SetTypes

Implementation roadmap
Phase 1 — Data Infrastructure (Weeks 1–2)
Provision Azure Synapse Analytics or SQL Server data warehouse. Create all 5 Fact and 5 Dimension tables per the star schema. Set up incremental refresh policies on FactSales, FactFinance, FactCustomer (daily). Configure row-level security (RLS) groups: C-Level, Regional VP, Department Head.
Phase 2 — Power Query & Data Model (Weeks 3–4)
Deploy all M code transforms in Power BI Desktop. Build DimDate using the provided M script. Create all relationships: Fact → Dim on surrogate keys (1:*). Validate cardinality. Disable auto date/time. Set storage mode to Import for Dims, DirectQuery for large Facts if > 50M rows.
Phase 3 — DAX Measures (Weeks 5–6)
Create a dedicated _Measures table (hidden). Implement all DAX measures in order of dependency: base aggregations first, then ratios, then composite indices. Use VAR/RETURN pattern throughout. Test each measure with field evaluation in a matrix visual before promoting to dashboard cards.
Phase 4 — Dashboard Build (Weeks 7–9)
Apply JSON theme file via View → Themes → Browse. Build all 6 pages per wireframe. Configure cross-filter interactions. Add Drill-Through pages for Customer Detail, Product Detail, Region Detail. Set up Bookmarks for saved filter states. Build navigation buttons using blank button + Bookmark action. Add dynamic titles using SELECTEDVALUE() in DAX.
Phase 5 — Performance Optimization
Run Performance Analyzer — target all visuals under 3s. Avoid CALCULATE inside iterators. Use SUMMARIZECOLUMNS over ADDCOLUMNS+SUMMARIZE. Enable query folding in Power Query. Partition large fact tables by year. Use aggregations on FactSales for high-cardinality columns. Remove unused columns from the model — every column costs memory.
Phase 6 — Governance & Security
Publish to Power BI Premium workspace. Configure RLS roles in Power BI Service. Enable sensitivity labels (Confidential — Executive Only) via Microsoft Purview. Set up workspace certification. Schedule dataset refresh via gateway (on-prem) or dataflow. Enable usage metrics dashboard for adoption tracking. Set up alerts on KPI cards for threshold breaches (e.g., churn > 8%). Document all measures in a Data Dictionary table within the .pbix.
Phase 7 — Deployment & Enablement
Use deployment pipelines (Dev → Test → Production). Configure scheduled refresh: FactSales hourly, FactFinance daily, FactCustomer daily. Embed in Teams or SharePoint for exec access. Create mobile layout for each page using Mobile Layout view. Conduct executive walkthrough and handover session. Set up a quarterly review cadence for KPI definition updates.
